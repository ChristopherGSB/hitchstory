-----------

PARAMETERIZED TESTING
---------------------

scenario:
  - click: test
  - fill form:
      first name: {{ first name }}
      last name: {{ last name }}
  - fill form:
      telephone: {{ telephone }}
  - do thing: {{ thistextwillappear }}
with:
  first name: Thomas
  last name: Edison
  telephone: 07771 1777124
